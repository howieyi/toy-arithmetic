### 题解
```javascript
/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum = function (candidates, target) {
  if (!candidates || !candidates.length || !target) return [];

  var result = [];
  var dfs = function (target, buffer, idx) {
    // 递归到数组边界则结束递归
    if (idx === candidates.length) return;

    // 当前结果集如果为 0 则寻找到合理结果集
    if (target === 0) {
      // 收集结果，结束递归
      result.push(buffer);
      return;
    }

    // 直接跳过当前值
    dfs(target, buffer, idx + 1);

    // 选择当前数
    // target - 当前索引数值 >= 0 则该值选入，继续递归寻找结果集
    if (target - candidates[idx] >= 0) {
      dfs(target - candidates[idx], [...buffer, candidates[idx]], idx);
    }
  };

  dfs(target, [], 0);
  return result;
};
```
